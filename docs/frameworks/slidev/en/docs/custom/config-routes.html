Configure Routes

Add your pages to the Slidev app.

Usage

Create ./setup/routes.ts with the following content:

import { defineRoutesSetup } from '@slidev/types'

export default defineRoutesSetup((routes) => {
  return [
    ...routes,
    {
      path: '/my-page',
      component: () => import('../pages/my-page.vue'),
    },
  ]
})
